// deadlock_avoided.c
#include <mpi.h>
#include <stdio.h>
int main(int argc, char *argv[]) {
int rank, size;
int msg = 100;
MPI_Init(&argc, &argv);
MPI_Comm_rank(MPI_COMM_WORLD, &rank);
MPI_Comm_size(MPI_COMM_WORLD, &size);
if (size < 2) {
if (rank == 0)
printf("This program requires at least 2 processes.\n");
MPI_Finalize();
return 0;
}
if (rank == 0) {
MPI_Send(&msg, 1, MPI_INT, 1, 0, MPI_COMM_WORLD); // Send first
MPI_Recv(&msg, 1, MPI_INT, 1, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE); // Then receive
} else if (rank == 1) {
MPI_Recv(&msg, 1, MPI_INT, 0, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE); // Receive first
MPI_Send(&msg, 1, MPI_INT, 0, 0, MPI_COMM_WORLD); // Then send
}
printf("Process %d completed communication safely.\n", rank);
MPI_Finalize();
return 0;
}
